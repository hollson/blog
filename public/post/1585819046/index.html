一. ZK集群连接对象

可以使用zk.WithXXX函数创建私有Option闭包对象，，然后以回调函数的方式传递(注入)给ZK连接方法。

// "github.com/sirupsen/logrus"
func Connect() *zk.Conn {
   // 集群
   var hosts = []string{"localhost:2181", "localhost:2182", "localhost:1234"}

   // WithXXX是一个私有闭包函数(Option)
   opt1 := zk.WithLogInfo(false)                       //是否开启日志
   opt2 := zk.WithLogger(logrus.New())                //自定义日志实现(github.com/sirupsen/logrus)
   opt3 := zk.WithHostProvider(&zk.DNSHostProvider{}) // 自定义Host程序，可更改集群主机读取方式

   conn, _, err := zk.Connect(hosts, 10*time.Second, opt1, opt2, opt3)
   if err != nil {
      fmt.Println(err)
   }
   return conn
}

package main

import (
   "fmt"
   "github.com/samuel/go-zookeeper/zk"
   "github.com/sirupsen/logrus"
   "time"
)

func main() {
   Create()
   Set()
   //Delete()
   Get()
   Children()
}

/**
*创建连接
 */
func Connect() *zk.Conn {
   var hosts = []string{"localhost:2181", "localhost:2182"}
   conn, _, err := zk.Connect(hosts, 10*time.Second,zk.WithLogInfo(false) )
   if err != nil {
      fmt.Println(err)
   }
   return conn
}

/**
* 创建节点
* 语法：create [-s] [-e] [-c] [-t ttl] path [data] [acl]
 */
func Create() {
   conn := Connect()
   defer conn.Close()

   var (
      path  = "/go"
      data  = []byte("hello golang")
      flags = 0                       //0:永久；1:临时；2:自增序号；3:临时自增
      acls  = zk.WorldACL(zk.PermAll) // 访问权限模式
   )
   //conn.Create("/gotest", nil, zk.FlagEphemeral, zk.WorldACL(zk.PermAll))
   pth, err := conn.Create(path, data, int32(flags), acls)
   if err != nil {
      panic(err)
   }
   fmt.Printf("创建成功: %v\n\n", pth)
}

/**
* 修改节点
* 语法：set [-s] [-v version] path data
*      version参数用于CAS支持,即乐观锁
 */
func Set() {
   conn := Connect()
   defer conn.Close()

   stat, err := conn.Set("/go", []byte("hi golang"), -1)
   if err != nil {
      panic(err)
   }
   fmt.Printf("修改成功: %+v\n\n", stat)
}

/**
* 删除节点 (不支持DeleteAll)
* 语法：delete [-v version] path
 */
func Delete() {
   conn := Connect()
   defer conn.Close()

   err := conn.Delete("/go", -1)
   if err != nil {
      panic(err)
   } else {
      fmt.Println("删除成功")
   }
}

/**
* 查询节点
 */
func Get() {
   conn := Connect()
   defer conn.Close()

   data, stat, err := conn.Get("/go")
   if err != nil {
      panic(err)
   }
   fmt.Printf("查询:%q,%+v\n\n", string(data), stat)
}

/**
* 查询子节点名称(不递归)
 */
func Children() {
   conn := Connect()
   defer conn.Close()

   children, _, err := conn.Children("/go")
   if err != nil {
      fmt.Println(err)
   }
   fmt.Printf("%+v \n", children)
}

package main

import (
   "fmt"
   "github.com/samuel/go-zookeeper/zk"
   "time"
)

func main() {
   PartlyWatch()
   fmt.Println()
   GlobalWatch()
}

// 局部监控
func PartlyWatch() {
   var (
      path        = "/partly_watch"
      hosts       = []string{"localhost:2181"}

      partlyEvent = func(evt <-chan zk.Event) {
         event := <-evt
         fmt.Println("*******************")
         fmt.Println("path:", event.Path)
         fmt.Println("type:", event.Type.String())
         fmt.Println("state:", event.State.String())
      }
   )

   conn, _, err := zk.Connect(hosts, time.Second, zk.WithLogInfo(false))
   defer conn.Close()
   if err != nil {
      panic(err)
   }

   _, _, event, err := conn.ExistsW(path)
   if err != nil {
      fmt.Println(err)
      return
   }

   // 异步监控
   go partlyEvent(event)

   // 创建节点
   create(conn, path)
   time.Sleep(time.Second * 1)
}

// 全局监控(事件
func GlobalWatch() {
   var (
      path  = "/global_watch"
      hosts = []string{"localhost:2181"}

      globalEvent = func(event zk.Event) {
         fmt.Println("*******************")
         fmt.Println("path:", event.Path)
         fmt.Println("type:", event.Type.String())
         fmt.Println("state:", event.State)
      }
   )

   opt1 := zk.WithLogInfo(false)
   opt2 := zk.WithEventCallback(globalEvent)
   conn, _, err := zk.Connect(hosts, time.Second, opt1, opt2)
   defer conn.Close()
   if err != nil {
      panic(err)
   }

   // 首次次监控
   is, stat, event, err := conn.ExistsW(path)
   if err != nil {
      panic(err)
   } else {
      fmt.Println(is, stat, event)
   }

   // 创建节点
   create(conn, path)
   time.Sleep(time.Second * 1)

   // 再次监控
   is, stat, event, err = conn.ExistsW(path)
   if err != nil {
      panic(err)
   } else {
      fmt.Println(is, stat, event)
   }
}

// 创建临时节点
func create(conn *zk.Conn, path string) {
   _, err := conn.Create(path, []byte("hello"), zk.FlagEphemeral, zk.WorldACL(zk.PermAll))
   if err != nil {
      panic(err)
   }
   fmt.Println("节点添加成功")
}
